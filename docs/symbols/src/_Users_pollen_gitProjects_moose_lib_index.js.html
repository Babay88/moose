1 var Client = require('mysql').Client,
  2         dataset = require("./dataset"),
  3         model = require("./model"),
  4         adapters = require("./adapters"),
  5         comb = require("comb"),
  6         hitch = comb.hitch,
  7         Promise = comb.Promise,
  8         PromiseList = comb.PromiseList,
  9         plugins = require("./plugins"),
 10         migration = require("./migrations"),
 11         Table = require("./table").Table;
 12 
 13 var connectionReady = false;
 14 
 15 /**
 16  * @class A singleton class that acts as the entry point for all actions performed in moose.
 17  *
 18  * @constructs
 19  * @name moose
 20  * @augments Migrations
 21  * @param options
 22  *
 23  */
 24 var Moose = comb.singleton(migration, {
 25     instance : {
 26     /**
 27      * @lends moose.prototype
 28      */
 29 
 30         /**
 31          * the models that are created and ready to be used.
 32          * @private
 33          */
 34         models : null,
 35 
 36         /**
 37          * The schemas that are created and ready to be used as the base of a model.
 38          * @private
 39          */
 40         schemas : null,
 41 
 42         /**
 43          *The type of database moose will be connecting to. Currently only mysql is supported.
 44          */
 45         type : "mysql",
 46 
 47 
 48         constructor : function(options) {
 49             this.__deferredSchemas = [];
 50             this.__deferredModels = [];
 51             this.models = {};
 52             this.schemas = {};
 53             this.super(arguments);
 54         },
 55 
 56         /**
 57          * Initialize the connection information, and prepare moose to communicate with the DB.
 58          * All models, and schemas, and models that are created before this method has been called will be deferred.
 59          *
 60          * @example
 61          *
 62          * moose.createConnection({
 63          *              type : "mysql",
 64          *              maxConnections : 1,
 65          *              minConnections : 1,
 66          *              user : "test",
 67          *              password : "testpass",
 68          *              database : 'test'
 69          *});
 70          *
 71          * @param {Object} options the options used to initialize the database connection.
 72          * @params {Number} [options.maxConnections = 10] the number of connections to pool.
 73          * @params {Number} [options.minConnections = 3] the number of connections to pool.
 74          * @param {String} [options.type = "mysql"] the type of database to communicate with.
 75          * @params {String} options.user the user to authenticate as.
 76          * @params {String} options.password the password of the user.
 77          * @params {String} options.database the name of the database to use.
 78          */
 79         createConnection : function(options) {
 80             //allow only one connection
 81             //We can  use the private singleton value, because moose is actually a singleton
 82             //TODO change this to allow connections to multiple databases
 83             if (!connectionReady) {
 84                 options = options || {};
 85                 this.options = options;
 86                 this.type = options.type || "mysql";
 87                 if (adapters[this.type]) {
 88                     this.client = new adapters[this.type].client(options);
 89                 } else {
 90                     throw new Error(this.type + " is not supported");
 91                 }
 92                 connectionReady = true;
 93             }
 94         },
 95 
 96         /**
 97          * Retrieves a connection to the database. Can be used to work with the database driver directly.
 98          *
 99          * @param {Boolean} autoClose if set to true then a new connection will be created,
100          *        otherwise a connection will be retrieved from a connection pool.
101          *
102          * @return {Query} a query object.
103          */
104         getConnection : function(autoClose) {
105             return this.client.getConnection(autoClose);
106         },
107 
108         /**
109          * Creates a context for performing database transactions.
110          * When using a transaction be sure to call commit with finished!
111          * @example
112          *
113          * var trans = moose.transaction();
114          * Do lots of stuff
115          * .
116          * .
117          * .
118          * trans.commit();
119          *
120          *
121          * @return {TransactionQuery} a transaction context.
122          */
123         transaction : function() {
124             return this.client.transaction();
125         },
126 
127         /**
128          * Creates a dataset to operate on a particular table.
129          *
130          * @param tableName the name of the table to perfrom operations on.
131          *
132          * @return {Dataset} a dataset to operate on a particular table.
133          */
134         getDataset : function(tableName) {
135             if (tableName) {
136                 return dataset.getDataSet(tableName, this.getConnection(),
137                         this.type);
138             } else {
139                 throw new Error("Table name required get getting a dataset");
140             }
141         },
142 
143         /**
144          * Execute raw SQL.
145          *
146          * @example
147          *  moose.execute("select * from myTable");
148          *
149          * @param sql the SQL to execute
150          *
151          * @returns {comb.Promise} a promise that will be called after the SQL execution completes.
152          */
153         execute : function(sql) {
154             var promise = new Promise();
155             var db = this.transaction();
156             db.query(sql).then(function(res) {
157                 db.commit().then(hitch(promise, "callback", res));
158             }, hitch(promise, "errback"));
159             return promise;
160         },
161 
162         /**
163          * Load a {@link moose.Table} to be used by a model or directly.
164          * This is typically called before, one creates a new model.
165          *
166          * @example
167          *
168          * moose.loadSchema("testTable").then(function(schema){
169          *     moose.addModel(schema, ...);
170          * });
171          *
172          * @param {String} tableName the name of the table to load
173          *
174          * @return {comb.Promise} A promise is called back with a table ready for use.
175          */
176         loadSchema : function(tableName) {
177             var promise = new Promise();
178             if (connectionReady) {
179                 if (!(tableName in this.schemas)) {
180                     adapters[this.type].schema(tableName, this.getConnection())
181                             .then(hitch(this, function(schema) {
182                         if (schema) {
183                             this.schemas[tableName] = schema;
184                         }
185                         promise.callback(schema);
186                     }), hitch(promise, "errback"));
187                 } else {
188                     promise.callback(this.schemas[tableName]);
189                 }
190             } else {
191                 this.__deferredSchemas.push(tableName);
192             }
193             return promise;
194         },
195 
196         /**
197          * Use to load a group of tables.
198          * @example
199          *  moose.loadSchema(["testTable", "testTable2", ....]).then(function(schema1, schema2,....){
200          *     moose.addModel(schema1, ...);
201          *     moose.addModel(schema2, ...);
202          * });
203          *
204          *
205          * @param {Array} tableNames the names of the tables to load.
206          *
207          * @return {comb.Promise} A promise that is called back with the tables in the same order that they were contained in the array.
208          */
209         loadSchemas : function(tableNames) {
210             var ret = new Promise();
211             if (tableNames instanceof Array) {
212                 if (tableNames.length) {
213                     var ps = tableNames.map(function(name) {
214                         return this.loadSchema(name);
215                     }, this);
216                     var pl = new PromiseList(ps);
217                     pl.addCallback(function(r) {
218                         // loop through and load the results
219                         ret.callback.apply(ret, r.map(function(o) {
220                             return o[1];
221                         }));
222                     });
223                     pl.addErrback(hitch(ret, "errback"));
224                 } else {
225                     ret.callback(null);
226                 }
227 
228             } else {
229                 throw new Error("tables names must be an array");
230             }
231             return ret;
232         },
233 
234         /**
235          * <p>Adds a model to moose.</p>
236          * </br>
237          * <b>NOTE</b>
238          * <ul>
239          *     <li>If a {moose.Table} is the first parameter then the {moose.Model} is returned immediately</li>
240          *     <li>If a table name is the first parameter then a {comb.Promise} is returned, and called back with the model once it is loaded</li>
241          * </ul>
242          *
243          * @example
244          *
245          *  moose.addModel(yourTable, moose, {
246          *      plugins : [PLUGIN1, PLUGIN2, PLUGIN3]
247          *      instance : {
248          *          myInstanceMethod : funciton(){},
249          *          getters : {
250          *              myProp : function(){
251          *                  return prop;
252          *              }
253          *          },
254          *
255          *          setters : {
256          *              myProp : function(val){
257          *                   prop = val;
258          *              }
259          *          }
260          *      },
261          *
262          *      static : {
263          *          myStaticMethod : function(){
264          *
265          *          },
266          *
267          *           getters : {
268          *              myStaticProp : function(){
269          *                  return prop;
270          *              }
271          *          },
272          *
273          *          setters : {
274          *              myStaticProp : function(val){
275          *                   prop = val;
276          *              }
277          *          }
278          *      },
279          *
280          *      pre : {
281          *          save : function(){
282          *
283          *          }
284          *      },
285          *
286          *      post : {
287          *          save  : function(){
288          *
289          *          }
290          *      }
291          *  });
292          *
293          *
294          * @param {String|moose.Table} table the table to be used as the base for this model.
295          * Factory for a new Model.
296          * @param {Object} options - Similar to {@link comb.define} with a few other conveniences
297          * @param {Array} options.plugins a list of plugins to enable on the model.
298          * @param {Object} options.pre an object containing key value pairs of events, and the corresponding callback.
299          *  <pre class="code">
300          *   {
301          *      pre : {
302          *          save : funciton(){},
303          *          update : function(){},
304          *          load : function(){},
305          *          remove : function(){}
306          *      }
307          *   }
308          * </pre>
309          *
310          * @param {Object} options.post an object containing key value pairs of events, and the corresponding callback.
311          *  <pre class="code">
312          *   {
313          *      post : {
314          *          save : funciton(){},
315          *          update : function(){},
316          *          load : function(){},
317          *          remove : function(){}
318          *      }
319          *   }
320          * </pre>
321          *
322          *
323          * @return {comb.Promise|Model} see description.
324          *
325          */
326         addModel : function(table, options) {
327             var promise = new Promise(), m;
328             if (table instanceof Table) {
329                 m = (this.models[table.tableName] = model.create(table, this,
330                         options || {}));
331                 return m;
332             } else {
333                 if (connectionReady) {
334                     if (this.models[table]) {
335                         promise.callback(this.models[table]);
336                     } else {
337                         if (typeof table == "string") {
338                             if (!table in this.schemas) {
339                                 this.loadSchema(table).then(
340                                         function(schema) {
341                                             var m = (this.models[table] = model
342                                                     .create(schema, this,
343                                                     options || {}));
344                                             promise.callback(m);
345                                         }, hitch(promise, "errback"));
346                             } else {
347                                 m = (this.models[table] = model.create(
348                                         this.schemas[table], this, options
349                                         || {}));
350                                 promise.callback(m);
351                             }
352                         }
353                     }
354                 } else {
355                     this.__deferredModels.push(arguments);
356                 }
357             }
358             return promise;
359         },
360 
361         /**
362          * Retrieve an already created table.
363          *
364          * @param {String} tableName the name of the table
365          *
366          * @return {moose.Table} return the table or null of it is not found.
367          */
368         getSchema : function(tableName) {
369             var m = null;
370             if (tableName in this.schemas) {
371                 m = this.schemas[tableName];
372             }
373             return m;
374         },
375 
376         /**
377          * Retrieve an already created model.
378          *
379          * @param {String} tableName the name of the table the model wraps.
380          *
381          * @return {moose.Model} return the model or null of it is not found.
382          */
383         getModel : function(tableName) {
384             var m = null;
385             if (this.models[tableName]) {
386                 m = this.models[tableName];
387             }
388             return m;
389         }
390     }
