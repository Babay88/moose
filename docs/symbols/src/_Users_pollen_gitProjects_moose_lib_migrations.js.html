 11 /**
 12  *@class This is a plugin for moose to add migration functionality.
 13  * Migrations are the preferred way of handling table creation, deletion, and altering.
 14  * <p>Adds
 15  * <ul>
 16  *     <li>migrate - perform a migration</li>
 17  *     <li>createTable - to create a new table</li>
 18  *     <li>dropTable - drop a table</li>
 19  *     <li>alterTable - alter an existing table</li>
 20  * </ul>
 21  *
 22  * Migrations are done through files contained in a particular directory, the directory should only contain migrations.
 23  * In order for moose to determine which versions to use
 24  * the file names must end in <versionNumber>.js where versionNumber is a integer value representing the version num.
 25  * An example directory structure might look like the following:
 26  *
 27  * <pre class="code">
 28  * -migrations
 29  *      - createFirstTables.1.js
 30  *      - shortDescription.2.js
 31  *      - another.3.js
 32  *      .
 33  *      .
 34  *      .
 35  *      -lastMigration.n.js
 36  * </pre>
 37  *
 38  * In order to easily identify where certain schema alterations have taken place it is a good idea to provide a brief
 39  * but meaningful migration name.
 40  *
 41  * createEmployee.1.js
 42  * </br>
 43  *
 44  * In order to run a migraton all one has to do is call moose.migrate(options);
 45  *
 46  * <pre class="code">
 47  *  moose.migrate({
 48  *               connection : {user : "test", password : "testpass", database : 'test'}, //Connection information to connect to the database.
 49  *               dir : "location of migrations", //Location of the directory containing the migrations.
 50  *               start : 0,//What version to start migrating at.
 51  *               end : 0,  //What version to stop migrations at.
 52  *               up : true //set to true to go up in migrations, false to rollback
 53  *  });
 54  * </pre>
 55  *
 56  * <p>Example migration file</b>
 57  * <pre class="code">
 58  *
 59  * //Up function used to migrate up a version
 60  * exports.up = function() {
 61  *   //create a new table
 62  *   moose.createTable("company", function(table) {
 63  *       //the table instance is passed in.
 64  *       //add columns
 65  *       table.column("id", types.INT({allowNull : false, autoIncrement : true}));
 66  *       table.column("companyName", types.VARCHAR({length : 20, allowNull : false}));
 67  *       //set the primary key
 68  *       table.primaryKey("id");
 69  *   });
 70  *   moose.createTable("employee", function(table) {
 71  *       table.column("id", types.INT({allowNull : false, autoIncrement : true}));
 72  *       table.column("firstname", types.VARCHAR({length : 20, allowNull : false}));
 73  *       table.column("lastname", types.VARCHAR({length : 20, allowNull : false}));
 74  *       table.column("midinitial", types.CHAR({length : 1}));
 75  *       table.column("gender", types.ENUM({enums : ["M", "F"], allowNull : false}));
 76  *       table.column("street", types.VARCHAR({length : 50, allowNull : false}));
 77  *       table.column("city", types.VARCHAR({length : 20, allowNull : false}));
 78  *       table.primaryKey("id");
 79  *   });
 80  *
 81  *   moose.createTable("companyEmployee", function(table) {
 82  *       table.column("companyId", types.INT({allowNull : false}));
 83  *       table.column("employeeId", types.INT({allowNull : false}));
 84  *       table.primaryKey(["companyId", "employeeId"]);
 85  *       table.foreignKey({companyId : {company : "id"}, employeeId : {employee : "id"}});
 86  *   });
 87  *   moose.createTable("works", function(table) {
 88  *       table.column("id", types.INT({allowNull : false, autoIncrement : true}));
 89  *       table.column("eid", types.INT({allowNull : false}));
 90  *       table.column("companyName", types.VARCHAR({length : 20, allowNull : false}));
 91  *       table.column("salary", types.DOUBLE({size : 8, digits : 2, allowNull : false}));
 92  *       table.primaryKey("id");
 93  *       table.foreignKey("eid", {employee : "id"});
 94  *
 95  *   });
 96  *};
 97  *
 98  * //Down function used to migrate down version
 99  *exports.down = function() {
100  *    moose.dropTable("companyEmployee");
101  *    moose.dropTable("works");
102  *    moose.dropTable("employee");
103  *    moose.dropTable("company");
104  *};
105  * </pre>
106  *@name Migrations
107  *
108  * */

114         constructor : function() {
115             this.__deferredMigrations = [];

118         /**
119          * Performs the deferred migrations function
120          *
121          * Used by create/drop/alter Table functions when within a migraton
122          * */
123         __migrateFun : function(fun, arg, conn) {
124             var ret = new Promise();
125             var adapter = adapters[this.type];
126             var connection = conn || this.getConnection();
127             adapter[fun](arg, conn).then(hitch(ret, "callback"), hitch(ret, "errback"));
128             return ret;
129         },

131         /**
132          * <p>Creates a new table. This function should be used while performing a migration.</p>
133          *
134          *
135          * @example : moose.createTable("test", function(table){
136          *     table.column("id", types.INT())
137          *     table.primaryKey("id");
138          * });
139          *
140          *
141          * @param {String} tableName the name of the table to create
142          * @param {Funciton} cb this funciton is callback with the table
143          *      - All table properties should be specified within this block
144          *
145          * @return {comb.Promise} There are two different results that the promise can be called back with.
146          * <ol>
147          *     <li>If a migration is currently being performed then the promise is called back with a
148          *     function that should be called to actually perform the migration.</li>
149          *     <li>If the called outside of a migration then the table is created immediately and
150          *     the promise is called back with the result.</li>
151          * </ol>
152          *
153          * */
154         createTable : function(tableName, cb) {
155             var table = new Table(tableName, {});
156             cb(table);
157             //add it to the moose schema map
158             this.schemas[tableName] = table;
159             if (!this.__inMigration) {
160                 return this.__migrateFun("createTable", table, this.getConnection());
161             } else {
162                 var ret = new Promise();
163                 ret.callback(hitch(this, "__migrateFun", "createTable", table));
164                 this.__deferredMigrations.push(ret);
165                 return ret;
166             }
167         },

170         /**
171          * Drops a table
172          *
173          * @example : moose.dropTable("test");
174          *
175          * @param {String} table the name of the table
176          *
177          * @return {comb.Promise} There are two different results that the promise can be called back with.
178          * <ol>
179          *     <li>If a migration is currently being performed then the promise is called back with a
180          *     function that should be called to actually perform the migration.</li>
181          *     <li>If the called outside of a migration then the table is dropped immediately and
182          *     the promise is called back with the result.</li>
183          * </ol>
184          **/
185         dropTable : function(table) {
186             table = new Table(table);
187             //delete from the moose schema map
188             delete this.schemas[table];
189             if (!this.__inMigration) {
190                 return this.__migrateFun("dropTable", "table", this.getConnection());
191             } else {
192                 var ret = new Promise();
193                 ret.callback(hitch(this, "__migrateFun", "dropTable", table));
194                 this.__deferredMigrations.push(ret);
195                 return ret;
196             }
197         },

199         /**
200          * Alters a table
201          *
202          * @example :
203          *
204          * moose.alterTable("test", function(table){
205          *     table.rename("test2");
206          *     table.addColumn("myColumn", types.STRING());
207          * });
208 
209          *
210          * @param {String} name The name of the table to alter.
211          * @param {Function} cb the function to execute with the table passed in as the first argument.
212          *
213          * @return {comb.Promise} There are two different results that the promise can be called back with.
214          * <ol>
215          *     <li>If a migration is currently being performed then the promise is called back with a
216          *     function that should be called to actually perform the migration.</li>
217          *     <li>If the called outside of a migration then the table is altered immediately and
218          *     the promise is called back with the result.</li>
219          * </ol>
220          * */
221         alterTable : function(name, cb) {
222             var ret = new Promise();
223             delete this.schemas[name];
224             if (!this.__inMigration) {
225                 this.loadSchema(name).then(function(table) {
226                     cb(table);
227                     this.__migrateFun("alterTable", table).then(hitch(ret, "callback"), hitch(ret, "errback"));
228                 }, hitch(ret, "errback"));
229             } else {
230                 this.loadSchema(name).then(hitch(this, function(table) {
231                     cb(table);
232                     ret.callback(hitch(this, "__migrateFun", "alterTable", table));
233                     delete this.schemas[name];
234                 }), hitch(ret, "errback"));
235                 this.__deferredMigrations.push(ret);
236                 return ret;
237             }
238         },

240         /**
241          * Performs the migration.
242          *
243          * @param {Function} fun the function to call to perform the migration
244          *                  this is an up or down function in a migration file, i.e. exports.up|down.
245          * @param {TransactionQuery} conn The current transaction, used when actually performing the database calls.
246          *
247          * @return {comb.Promise} called back after the migration completes, and the next can continue.
248          */
249         __doMigrate : function(fun, conn) {
250             var ret = new Promise();
251             this.__inMigration = true;
252             fun();
253             //all calls should be deferred
254             if (this.__deferredMigrations.length) {
255                 var defList = new PromiseList(this.__deferredMigrations);
256                 defList.then(hitch(this, function(res) {
257                     //map my to retrieve the actual responses
258                     //and call the associated method
259                     var ps = [], f;
260                     for (var i = 0, len = res.length; i < len; i++) {
261                         f = res[i][1];
262                         if (typeof  f == "function") {
263                             ps[i] = f(conn);
264                         } else {
265                             //todo move this to a common error handler like moose.error
266                             throw new Error("Unexpected error");
267                         }
268                     }
269                     //listen for the promises to complete
270                     new PromiseList(ps).then(hitch(ret, "callback"), hitch(ret, "errback"));
271                     //reset for another migration
272                     this.__deferredMigrations.length = 0;
273                 }), function(err) {
274                     for (var len = err.length, i = len - 1; i > 0; i--) {
275                         console.error(err[i]);
276                     }
277                     conn.rollback(hitch(ret, "errback", err));
278                 });
279             } else {
280                 promise.callback();
281             }
282             this.__inMigration = false;
283             return ret;
284         },

286         /**
287          * Perform a migration.
288          *
289          * @example
290          *  moose.migrate({
291          *               connection : {user : "test", password : "testpass", database : 'test'}, //Connection information to connect to the database.
292          *               dir : "location of migrations", //Location of the directory containing the migrations.
293          *               start : 0,//What version to start migrating at.
294          *               end : 0,  //What version to stop migrations at.
295          *               up : true //set to true to go up in migrations, false to rollback
296          *  });
297 
298          *
299          * <p><b>NOTE</b></br>
300          *  If you start at 0 and end at 0 your migrations will inlude the file only at "migrationName".0.js
301          *  if you specify start : 0 and end 1 then your migrations will include files "migraiton0".0.js and "migration1".1.js,
302          *  the names being whatever you specify before the *.0 and *.1
303          *  </p>
304 
305          *
306          * <p>Example migration file</b>
307          * @example
308          *
309          * //Up function used to migrate up a version
310          * exports.up = function() {
311          *   //create a new table
312          *   moose.createTable("company", function(table) {
313          *       //the table instance is passed in.
314          *       //add columns
315          *       table.column("id", types.INT({allowNull : false, autoIncrement : true}));
316          *       table.column("companyName", types.VARCHAR({length : 20, allowNull : false}));
317          *       //set the primary key
318          *       table.primaryKey("id");
319          *   });
320          *   moose.createTable("employee", function(table) {
321          *       table.column("id", types.INT({allowNull : false, autoIncrement : true}));
322          *       table.column("firstname", types.VARCHAR({length : 20, allowNull : false}));
323          *       table.column("lastname", types.VARCHAR({length : 20, allowNull : false}));
324          *       table.column("midinitial", types.CHAR({length : 1}));
325          *       table.column("gender", types.ENUM({enums : ["M", "F"], allowNull : false}));
326          *       table.column("street", types.VARCHAR({length : 50, allowNull : false}));
327          *       table.column("city", types.VARCHAR({length : 20, allowNull : false}));
328          *       table.primaryKey("id");
329          *   });
330          *
331          *   moose.createTable("companyEmployee", function(table) {
332          *       table.column("companyId", types.INT({allowNull : false}));
333          *       table.column("employeeId", types.INT({allowNull : false}));
334          *       table.primaryKey(["companyId", "employeeId"]);
335          *       table.foreignKey({companyId : {company : "id"}, employeeId : {employee : "id"}});
336          *   });
337          *   moose.createTable("works", function(table) {
338          *       table.column("id", types.INT({allowNull : false, autoIncrement : true}));
339          *       table.column("eid", types.INT({allowNull : false}));
340          *       table.column("companyName", types.VARCHAR({length : 20, allowNull : false}));
341          *       table.column("salary", types.DOUBLE({size : 8, digits : 2, allowNull : false}));
342          *       table.primaryKey("id");
343          *       table.foreignKey("eid", {employee : "id"});
344          *
345          *   });
346          *};
347          *
348          * //Down function used to migrate down version
349          *exports.down = function() {
350          *    moose.dropTable("companyEmployee");
351          *    moose.dropTable("works");
352          *    moose.dropTable("employee");
353          *    moose.dropTable("company");
354          *};
355          *
356          * @param {Object} options the options to specify how to perform the migration.
357          * @param {Object} options.connection : @see moose.createConnection
358          * @param {String} options.dir Location of the directory where migrations are located.
359          * @param {Boolean} [options.up = true] If true will migrate up otherwise down.
360          * @param {Number} [options.start = 0] The migration to start at.
361          * @param {Number} [options.end=Infinity] The migration to end at this, end is inclusive
362          *
363          * @return {comb.Promise} Called back after all migrations have completed.
364          */
365         migrate : function(options) {
366             var promise = new Promise();
367             var dir, file, migrationDir = comb.isBoolean(options.up) ? options.up : true;
368             var start = typeof options.end == "number" ? options.start : 0,
369                     end = typeof options.end == "number" ? options.end : Infinity;
370 
371             //if a coonection is not provided or we dont already have a connection
372             //throw an error
373             if (options.connection || this.__connectionReady) {
374                 //set our current state, allows us to push any create/alter/delete
375                 //transactions into our deferred
376 
377                 //init our connecton information
378                 !this.__connectionReady && this.createConnection(options.connection);
379                 if (options.dir) dir = path.resolve(process.cwd(), options.dir);
380                 //read the directory
381                 fs.readdir(dir, hitch(this, function(err, files) {
382                     var funs = [], fun = migrationDir ? "up" : "down", parts, num, file, extName = path.extname, baseName = path.basename;
383                     for (var i = 0, len = files.length; i < len; i++) {
384                         file = files[i];
385                         if (extName(file) == ".js") {
386                             //its a js file otherwise ignore
387                             //split the path to get the num
388                             parts = baseName(file, ".js").split(".");
389                             //get the migration number
390                             num = parseInt(parts.pop());
391                             //if it is a valid index
392                             if (!isNaN(num) && num >= start && num <= end) {
393                                 var cls = require(path.resolve(dir, file));
394                                 if (cls[fun]) {
395                                     //add it
396                                     funs[num - start] = hitch(cls, fun);
397                                 }
398                             }
399                         }
400                     }
401                     if (!migrationDir) {
402                         funs.reverse();
403                     }
404                     var conn = this.transaction();
405                     if (conn) {
406                         if (funs.length) {
407                             i = 0,len = funs.length;
408                             var doMig = hitch(this, this.__doMigrate);
409 
410                             var next = hitch(this, function(res) {
411                                 if (i < len) {
412                                     doMig(funs[i], conn).then(next, function(err) {
413                                         console.log(err);
414                                         conn.rollback(hitch(promise, "errback", err));
415                                     });
416                                 } else {
417                                     conn.commit().then(hitch(promise, "callback"));
418                                 }
419                                 i++;
420                             });
421                             next();
422                         } else {
423                             promise.callback();
424                         }
425 
426 
427                     }
428                 }));
429             } else {
430                 throw new Error("when migrating a connection is required");
431             }
432             return promise;
433         }
